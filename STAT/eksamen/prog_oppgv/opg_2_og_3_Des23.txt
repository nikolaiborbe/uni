import numpy as np

def simX(theta: float, n: int):
    U = np.random.uniform(0,1, size=n)
    X = np.sqrt(theta*np.ln(U-1))
    return X

def simP(Z_obs, n, m):    
    
    Theta_0 = 100
    z_count = 0

    for i in range(m):
        x_sim = simX(Theta_0, n)
        Theta_hat = sum(x_sim**2) / n
        z_sim = (Theta_hat - Theta_0) / np.sqrt(Theta_0**2 / n)

        if z_sim > Z_obs: z_count += 1

    return z_count / m
